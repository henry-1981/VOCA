"use client";

import { useEffect } from "react";

type Props = {
  newLevel: number;
  onClose: () => void;
};

export function LevelUpOverlay({ newLevel, onClose }: Props) {
  useEffect(() => {
    const timer = setTimeout(onClose, 3000);
    return () => clearTimeout(timer);
  }, [onClose]);

  return (
    <div
      className="fixed inset-0 z-[60] flex items-center justify-center bg-black/60"
      onClick={onClose}
    >
      {/* Gold expanding circle */}
      <div className="absolute w-64 h-64 rounded-full bg-gradient-radial from-amber-400/40 to-transparent animate-level-up" />

      <div className="text-center z-10 animate-scale-bounce">
        <div className="text-2xl font-bold text-amber-200 mb-2">Level Up!</div>
        <div className="text-8xl font-black text-amber-400 drop-shadow-[0_0_20px_rgba(251,191,36,0.5)]">
          {newLevel}
        </div>
      </div>
    </div>
  );
}

function getTimeOfDayTheme(): { gradient: string; starOpacity: string } {

  if (hour >= 5 && hour < 7) return { gradient: "from-pink-300 via-orange-200 to-sky-200", starOpacity: "opacity-20" };
  if (hour >= 7 && hour < 12) return { gradient: "from-sky-300 via-blue-200 to-amber-100", starOpacity: "opacity-0" };
  if (hour >= 12 && hour < 17) return { gradient: "from-sky-400 via-blue-300 to-sky-200", starOpacity: "opacity-0" };
  if (hour >= 17 && hour < 20) return { gradient: "from-purple-600 via-orange-400 to-amber-300", starOpacity: "opacity-60" };
  return { gradient: "from-indigo-950 via-purple-900 to-indigo-900", starOpacity: "opacity-100" };
}
